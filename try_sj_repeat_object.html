Looping with objects:

<script>
var names=[
{name:'Jani',country:'Norway'},
{name:'Hege',country:'Sweden'},
{name:'Kai',country:'Denmark'}]
</script>
<body>

<div sj-app="">

<ul>
  <li sj-repeat="x in names">

</ul>
